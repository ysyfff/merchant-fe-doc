Pagination_

根据UI图定制化Pagination，用法和Pagination基本一样

使用

export default class Demo extends React.Component{
  render(){
    return(
      <Pagination_ 
        defaultCurrent={1} 
        current={search.page} 
        onJump={page => search.page = page} 
        total={listData.totalCount} 
        onChange={this.handlePageChange}
      />
    )
  }
}
